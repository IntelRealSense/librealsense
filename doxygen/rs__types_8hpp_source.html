167  inline void error::handle(rs2_error* e)
168  {
169  if (e)
170  {
171  auto h = rs2_get_librealsense_exception_type(e);
172  switch (h) {
173  case RS2_EXCEPTION_TYPE_CAMERA_DISCONNECTED:
174  throw camera_disconnected_error(e);
175  case RS2_EXCEPTION_TYPE_BACKEND:
176  throw backend_error(e);
177  case RS2_EXCEPTION_TYPE_INVALID_VALUE:
178  throw invalid_value_error(e);
179  case RS2_EXCEPTION_TYPE_WRONG_API_CALL_SEQUENCE:
180  throw wrong_api_call_sequence_error(e);
181  case RS2_EXCEPTION_TYPE_NOT_IMPLEMENTED:
182  throw not_implemented_error(e);
183  case RS2_EXCEPTION_TYPE_DEVICE_IN_RECOVERY_MODE:
184  throw device_in_recovery_mode_error(e);
185  default:
186  throw error(e);
187  }
188  }
189  }
