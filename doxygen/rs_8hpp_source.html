130  /*
131  Your callback should look like this, for example:
132  void callback( rs2_log_severity severity, rs2::log_message const & msg ) noexcept
133  {
134  std::cout << msg.build() << std::endl;
135  }
136  and, when initializing rs2:
137  rs2::log_to_callback( callback );
138  or:
139  rs2::log_to_callback(
140  []( rs2_log_severity severity, rs2::log_message const & msg ) noexcept
141  {
142  std::cout << msg.build() << std::endl;
143  })
144  */
145  inline void log_to_callback( rs2_log_severity min_severity, log_callback::log_fn callback )
146  {
147  // We wrap the callback with an interface and pass it to librealsense, who will
148  // now manage its lifetime. Rather than deleting it, though, it will call its
149  // release() function, where (back in our context) it can be safely deleted:
150  rs2_error* e = nullptr;
151  rs2_log_to_callback_cpp( min_severity, new log_callback( std::move( callback )), &e );
152  error::handle( e );
153  }
